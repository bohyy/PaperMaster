# 绘顶代码模板

## 📊 折线图（Line Plot）

### Python (Matplotlib)

```python
import matplotlib.pyplot as plt
import numpy as np
from matplotlib import rcParams

# 设置学术风格
rcParams['font.family'] = 'Times New Roman'
rcParams['font.size'] = 10
rcParams['axes.labelsize'] = 10
rcParams['xtick.labelsize'] = 9
rcParams['ytick.labelsize'] = 9
rcParams['legend.fontsize'] = 9
rcParams['figure.dpi'] = 300

# 色盲友好型配色
colors = ['#0072B2', '#D55E00', '#009E73', '#F0E442', '#CC79A7']

# 数据（示例）
x = np.linspace(0, 10, 100)
y1 = np.sin(x)
y2 = np.cos(x)
y3 = np.sin(x) * np.cos(x)

# 创建图形
fig, ax = plt.subplots(figsize=(6, 4))

# 绘制线条
line1, = ax.plot(x, y1, color=colors[0], linewidth=2, label='Method A')
line2, = = ax.plot(x, y2, color=colors[1], linewidth=2, label='Method B')
line3, = ax.plot(x, y3, color=colors[2], linewidth=2, linestyle='--', label='Method C')

# 添加误差带（如果需要）
ax.fill_between(x, y1-0.1, y1+0.1, alpha=0.2, color=colors[0])

# 设置坐标轴
ax.set_xlabel('Time (s)', fontweight='bold')
ax.set_ylabel('Amplitude', fontweight='bold')
ax.set_title('Comparison of Three Methods', fontsize=12, fontweight='bold')

# 添加网格
ax.grid(True, linestyle='--', alpha=0.5, linewidth=0.5)

# 设置坐标轴范围
ax.set_xlim(0, 10)
ax.set_ylim(-1.5, 1.5)

# 添加图例
ax.legend(loc='best', frameon=True, fancybox=False, shadow=False)

# 调整布局
plt.tight_layout()

# 保存
plt.savefig('figure1.pdf', format='pdf', bbox_inches='tight')
plt.savefig('figure1.png', format='png', dpi=300, bbox_inches='tight')
plt.close()
```

### R (ggplot2)

```r
library(ggplot2)
library(cowplot)  # 用于组合图表

# 色盲友好型配色
cb_colors <- c(
  "#0072B2", "#D55E00", "#009E73",
  "#F0E442", "#CC79A7", "#56B4E9"
)

# 创建示例数据
data <- data.frame(
  x = rep(1:10, 3),
  y = c(
    sin(1:10),
    cos(1:10),
    sin(1:10) * cos(1:10)
  ),
  method = rep(c("Method A", "Method B", "Method C"), each=10)
)

# 绘制折线图
p <- ggplot(data, aes(x=x, y=y, color=method, group=method)) +
  geom_line(size=1) +
  geom_point(size=2, shape=16) +
  scale_color_manual(values=cb_colors[1:3]) +
  labs(
    title="Comparison of Three Methods",
    x="Time (s)",
    y="Amplitude"
  ) +
  theme_bw() +
  theme(
    plot.title = element_text(size=12, face="bold", hjust=0.5),
    axis.title = element_text(size=10, face="bold"),
    axis.text = element_text(size=9),
    legend.position = "right",
    legend.title = element_blank(),
    legend.text = element_text(size=9),
    panel.grid = element_line(linetype="dashed", size=0.5)
  )

# 保存
ggsave("figure1.pdf", p, width=6, height=4, device="pdf")
ggsave("figure1.png", p, width=6, height=4, dpi=300)
```

## 📊 某状图（Bar Plot）

### Python (Matplotlib)

```python
import matplotlib.pyplot as plt
import numpy as np

# 数据
categories = ['Method A', 'Method B', 'Method C', 'Method D']
values = [85, 72, 91, 68]
errors = [5, 4, 6, 5]  # 标准差

colors = ['#0072B2', '#D55E00', '#009E73', '#F0E442']

# 创建图形
fig, ax = plt.subplots(figsize=(8, 5))

# 绘制柱状图
bars = ax.bar(categories, values, yerr=errors,
              capsize=5, capthick=2,
              color=colors, alpha=0.8, edgecolor='black', linewidth=1.5)

# 在柱子上添加数值标签
for bar, value in zip(bars, values):
    height = bar.get_height()
    ax.text(bar.get_x() + bar.get_width()/2., height,
            f'{value:.1f}',
            ha='center', va='bottom', fontsize=9)

# 设置坐标轴
ax.set_xlabel('Methods', fontsize=10, fontweight='bold')
ax.set_ylabel('Accuracy (%)', fontsize=10, fontweight='bold')
ax.set_title('Performance Comparison', fontsize=12, fontweight='bold')

# 添加网格
ax.grid(True, linestyle='--', alpha=0.5, linewidth=0.5, axis='y')

# 设置 Y 轴范围
ax.set_ylim(0, 100)

# 调整布局
plt.tight_layout()

# 保存
plt.savefig('figure2.pdf', format='pdf', bbox_inches='tight')
plt.savefig('figure2.png', format='png', dpi=300, bbox_inches='tight')
plt.close()
```

## 🔴 散点图（Scatter Plot）

### Python (Matplotlib)

```python
import matplotlib.pyplot as plt
import numpy as np
from scipy import stats

# 生成示例数据
np.random.seed(42)
x = np.random.randn(100)
y = 0.5 * x + np.random.randn(100) * 0.5

# 计算拟合线和置信区间
slope, intercept, r_value, p_value, std_err = stats.linregress(x, y)
x_line = np.array([x.min(), x.max()])
y_line = slope * x_line + intercept

# 创建图形
fig, ax = plt.subplots(figsize=(6, 5))

# 绘制散点
scatter = ax.scatter(x, y, c='#0072B2', alpha=0.6,
                    s=50, edgecolors='black', linewidth=0.5)

# 绘制拟合线
ax.plot(x_line, y_line, color='#D55E00', linewidth=2,
        label=f'y = {slope:.2f}x + {intercept:.2f}')

# 添加 R² 标注
r_squared = r_value ** 2
ax.text(0.05, 0.95, f'R² = {r_squared:.3f}',
        transform=ax.transAxes, fontsize=10,
        verticalalignment='top', bbox=dict(boxstyle='round', facecolor='white', alpha=0.8))

# 设置坐标轴
ax.set_xlabel('X Variable', fontsize=10, fontweight='bold')
ax.set_ylabel('Y Variable', fontsize=10, fontweight='bold')
ax.set_title('Scatter Plot with Linear Fit', fontsize=12, fontweight='bold')

# 添加网格
ax.grid(True, linestyle='--', alpha=0.5, linewidth=0.5)

# 添加图例
ax.legend(loc='best', fontsize=9)

plt.tight_layout()
plt.savefig('figure3.pdf', format='pdf', bbox_inches='tight')
plt.savefig('figure3.png', format='png', dpi=300, bbox_inches='tight')
plt.close()
```

## 🔥 热力图（Heatmap）

### Python (Matplotlib + Seaborn)

```python
import matplotlib.pyplot as plt
import seaborn as sns
import numpy as np

# 生成相关矩阵
np.random.seed(42)
data = np.random.randn(10, 10)
correlation_matrix = np.corrcoef(data)

# 创建图形
fig, ax = plt.subplots(figsize=(8, 7))

# 绘制热力图
sns.heatmap(correlation_matrix,
            annot=True,      # 显示数值
            fmt='.2f',       # 数值格式
            cmap='RdBu_r',   # 蓝红渐变（色盲友好）
            center=0,        # 中心值为 0
            vmin=-1, vmax=1, # 值范围
            square=True,     # 方形单元格
            cbar_kws={'label': 'Correlation Coefficient'},
            linewidths=0.5,  # 单元格边框
            ax=ax)

# 设置标题
ax.set_title('Correlation Matrix', fontsize=12, fontweight='bold', pad=20)

plt.tight_layout()
plt.savefig('figure4.pdf', format='pdf', bbox_inches='tight')
plt.savefig('figure4.png', format='png', dpi=300, bbox_inches='tight')
plt.close()
```

## 📦 箱线图（Box Plot）

### Python (Matplotlib)

```python
import matplotlib.pyplot as plt
import numpy as np

# 生成示例数据
np.random.seed(42)
data = {
    'Group A': np.random.normal(10, 2, 100),
    'Group B': np.random.normal(12, 1.5, 100),
    'Group C': np.random.normal(8, 2.5, 100),
    'Group D': np.random.normal(11, 1.8, 100)
}

# 准备数据
groups = list(data.keys())
values = list(data.values())

colors = ['#0072B2', '#D55E00', '#009E73', '#F0E442']

# 创建图形
fig, ax = plt.subplots(figsize=(8, 6))

# 绘制箱线图
bp = ax.boxplot(values, labels=groups,
                patch_artist=True,  # 允许填充颜色
                widths=0.6,
                showmeans=True,     # 显示均值
                meanprops={'marker':'D', 'markerfacecolor':'black', 'markersize':5})

# 设置颜色
for patch, color in zip(bp['boxes'], colors):
    patch.set_facecolor(color)
    patch.set_alpha(0.7)

# 设置坐标轴
ax.set_xlabel('Groups', fontsize=10, fontweight='bold')
ax.set_ylabel('Values', fontsize=10, fontweight='bold')
ax.set_title('Distribution Comparison', fontsize=12, fontweight='bold')

# 添加网格
ax.grid(True, linestyle='--', alpha=0.5, linewidth=0.5, axis='y')

# 添加显著性标记（示例）
y_max = max([max(v) for v in values])
ax.text(1, y_max + 1, '*', fontsize=20, ha='center')
ax.plot([1, 2], [y_max + 0.8, y_max + 0.8], 'k-', linewidth=1)

plt.tight_layout()
plt.savefig('figure5.pdf', format='pdf', bbox_inches='tight')
plt.savefig('figure5.png', format='png', dpi=300, bbox_inches='tight')
plt.close()
```

## 💡 使用提示

### 通用设置

```python
# 设置学术风格（全局）
def setup_academic_style():
    """设置 Matplotlib 学术风格"""
    import matplotlib as mpl

    # 字体设置
    mpl.rcParams['font.family'] = 'Times New Roman'
    mpl.rcParams['font.size'] = 10

    # 坐标轴设置
    mpl.rcParams['axes.labelsize'] = 10
    mpl.rcParams['axes.titlesize'] = 12
    mpl.rcParams['xtick.labelsize'] = 9
    mpl.rcParams['ytick.labelsize'] = 9

    # 线条设置
    mpl.rcParams['lines.linewidth'] = 1.5
    mpl.rcParams['lines.markersize'] = 6

    # 图例设置
    mpl.rcParams['legend.fontsize'] = 9
    mpl.rcParams['legend.frameon'] = True

    # 图片设置
    mpl.rcParams['figure.dpi'] = 300
    mpl.rcParams['savefig.dpi'] = 300
    mpl.rcParams['savefig.bbox'] = 'tight'
    mpl.rcParams['savefig.format'] = 'pdf'

# 使用
setup_academic_style()
```
